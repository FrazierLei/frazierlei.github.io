PATA-1077 Kuchiguse

August 5, 2021

题目链接

本质是求若干个字符串的公共最长后缀，可以通过反转转化成求最长前缀的问题。

在 leetcode 上看到了这个精炼的解答——排序、比较首尾两个字符串。

#include <bits/stdc++.h>
using namespace std;
using gg = long long;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    gg n;
    cin >> n;
    cin.get();
    string s;
    vector<string> v;
    while (n--){
        getline(cin, s);
        reverse(s.begin(), s.end());
        v.push_back(s);
    }
    sort(v.begin(), v.end());
    string a = v.front(), b = v.back(), c;
    gg i;
    for (i = 0; a[i] == b[i] and (i < min(a.size(), b.size())); ++i);
    if (i == 0)
        cout << "nai";
    else{
        c = a.substr(0, i);
        reverse(c.begin(), c.end());
        cout << c;
    }
    return 0;
}
