PAT 排名问题

August 7, 2021

同样成绩的要有同样的排名。

代码模板

vector<gg> grade = {98, 97, 93, 93, 87, 84, 84, 84, 80};
vector<gg> rank(grade.size());

for (gg i = 0, r = 1; i < grade.size(); ++i){
  if (i == 0 or grade[i] != grade[i - 1])
    r = i + 1;
  rank[i] = r;
}

PATA-1025

我用了map，没有定义函数，比答案稍微麻烦一点，但是不容易错。

#include <bits/stdc++.h>
using namespace std;
using gg = long long;
struct Student{
    string id;
    gg grade;
    Student(string i, gg g):id(i), grade(g){}
};

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    string id;
    gg n, k, grade;
    cin >> n;
    vector<Student> all;
    map<string, gg> local_num;
    map<string, gg> local_rank;
    map<string, gg> final_rank;
    for (gg i = 1; i <= n; ++i){
        cin >> k;
        vector<Student> group;
        while (k--){
            cin >> id >> grade;
            group.push_back(Student(id, grade));
            all.push_back(Student(id, grade));
            local_num.insert({id, i});
        }
        sort(group.begin(), group.end(), [](const Student& s1, const Student& s2){
            return tie(s2.grade, s1.id) < tie(s1.grade, s2.id);
        });
        for (gg j = 0, r = 1; j < group.size(); ++j){
            if (j == 0 or group[j].grade != group[j - 1].grade)
                r = j + 1;
            local_rank.insert({group[j].id, r});
        }
    }
    sort(all.begin(), all.end(), [](const Student& s1, const Student& s2){
        return tie(s2.grade, s1.id) < tie(s1.grade, s2.id);
    });
    for (gg j = 0, r = 1; j < all.size(); ++j){
        if (j == 0 or all[j].grade != all[j - 1].grade)
            r = j + 1;
        final_rank.insert({all[j].id, r});
    }
    cout << all.size() << "\n";
    for (auto i : all)
        cout << i.id << " " << final_rank[i.id] << " " << local_num[i.id] << " " << local_rank[i.id] << "\n";
    return 0;
}

答案：

#include <bits/stdc++.h>
using namespace std;
using gg = long long;
struct Student {
    gg id, ge, gi, gf;  // gf表示最终成绩
    array<gg, 6> app;  //申请的学校
};
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    gg ni, mi, ki;
    cin >> ni >> mi >> ki;
    vector<gg> quota(mi);
    for (gg& i : quota) {  //读取每个学校的名额数量
        cin >> i;
    }
    vector<Student> stu(ni);
    for (gg i = 0; i < ni; ++i) {  //读取每个学生的申请
        cin >> stu[i].ge >> stu[i].gi;
        for (gg j = 0; j < ki; ++j) {
            cin >> stu[i].app[j];
        }
        stu[i].gf = stu[i].ge + stu[i].gi;  //最终成绩不除2不会影响结果
        stu[i].id = i;
    }
    sort(stu.begin(), stu.end(), [](const Student& s1, const Student& s2) {
        return tie(s2.gf, s2.ge) < tie(s1.gf, s1.ge);
    });  //排序
    vector<gg> rank(ni);  //记录学生排名
    gg r = 1;
    for (gg i = 0; i < stu.size(); ++i) {  //计算排名
        if (i == 0 or stu[i].gf != stu[i - 1].gf or
            stu[i].ge != stu[i - 1].ge) {
            r = i + 1;
        }
        rank[stu[i].id] = r;
    }
    vector<vector<gg>> ans(mi);
    for (auto& s : stu) {  //统计每个学生的志愿录取情况
        for (gg i = 0; i < ki; ++i) {
            gg j = s.app[i];
            if (ans[j].size() < quota[j] or
                (not ans.empty() and rank[s.id] == rank[ans[j].back()])) {
                ans[j].push_back(s.id);
                break;
            }
        }
    }
    for (auto& v : ans) {  //输出
        sort(v.begin(), v.end());  //学生编号要升序输出
        for (gg i = 0; i < v.size(); ++i) {
            cout << (i == 0 ? "" : " ") << v[i];
        }
        cout << "\n";
    }
    return 0;
}
