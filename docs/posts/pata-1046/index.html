PAT A-1046 Shortest Distance（容易超时）

August 8, 2021

题目链接

本来觉得5分钟就能过，但是最后一个测试点过不去，超时。

#include <bits/stdc++.h>
using namespace std;
using gg = long long;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    gg n, k;
    cin >> n;
    vector<gg> v(n);
    for (gg &i : v)
        cin >> i;
    gg sum = accumulate(v.begin(), v.end(), 0);
    cin >> k;
    while (k--){
        gg i, j, d1, d2;
        cin >> i >> j;
        if (i > j)
            swap(i, j);
        d1 = accumulate(v.begin() + i - 1, v.begin() + j - 1, 0);
        d2 = sum - d1;
        cout << min(d1, d2) << "\n";
    }
    return 0;
}

容易发现其实做了很多重复的运算，答案如下：

用到了 partial_sum，v[i] 表示第一个出口到第 i+1 个出口的距离。

#include <bits/stdc++.h>
using namespace std;
using gg = long long;
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    gg ni, mi, sum = 0;
    cin >> ni;
    vector<gg> v(ni + 1);  //注意v[0]=0
    for (gg i = 1; i <= ni; ++i) {
        cin >> v[i];
    }
    partial_sum(v.begin(), v.end(), v.begin());  //求前缀和
    cin >> mi;
    while (mi--) {
        gg ai, bi;
        cin >> ai >> bi;
        gg t = v[max(ai, bi) - 1] - v[min(ai, bi) - 1];
        cout << min(t, v.back() - t) << "\n";
    }
    return 0;
}
