August 1, 2021

题目链接

最简单的方法肯定是将两个数组合并，然后排序，最后输出中位数。这样的时间复杂度是 $O\left((m+n)\log(m+n)\right)$

#include <bits/stdc++.h>
using namespace std;
using gg = long long;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    gg n, m;
    cin >> n;
    vector<gg> v(n);
    for (gg i = 0; i < n; ++i)
        cin >> v[i];
    cin >> m;
    v.resize(n + m);
    for (gg i = 0; i < m; ++i)
        cin >> v[n + i];
    sort(v.begin(), v.end());
    cout << v[(n + m - 1)/2];
    return 0;
}

如果要求时间复杂度为 $O(\log(m+n))$ 的话，就不能借助排序了，这里有一个很通俗易懂的算法解释：

漫画：如何找到两个数组的中位数？（修订版）

我的实现：

#include <bits/stdc++.h>
using namespace std;
using gg = long long;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    gg n, m;
    cin >> n;
    vector<gg> a(n);
    for (gg i = 0; i < n; ++i)
        cin >> a[i];
    cin >> m;
    vector<gg> b(m);
    for (gg i = 0; i < m; ++i)
        cin >> b[i];
    // 特殊情况1，a中最小元素大于b中最大元素
    gg t = (m + n - 1) / 2;
    if (*min_element(a.begin(), a.end()) > *max_element(b.begin(), b.end())){
        if (b.size() > t){
            cout << b[t];
            return 0;
        }
        else{
            cout << a[t - b.size()];
            return 0;
        }
    }
    // 特殊情况2，a中最大元素小于b中最小元素
    if (*max_element(a.begin(), a.end()) < *min_element(b.begin(), b.end())){
        if (a.size() > t){
            cout << a[t];
            return 0;
        }
        else{
            cout << b[t - a.size()];
            return 0;
        }
    }

    gg i = n / 2, j = (m + n + 1) / 2 - i;
    while (i >= 0 and i <= n and j >= 0 and j <= m){
        j = (m + n + 1) / 2 - i;
        if (a[i - 1] <= b[j] and b[j - 1] <= a[i])
            break;
        else if (a[i - 1] > b[j])
            --i;
        else if (b[j - 1] > a[i])
            ++i;
    }
    if ((n + m) / 2)
        cout << max(*max_element(a.begin(), a.begin() + i), max_element(b.begin(), b.begin() + j));
    else
        cout << min(*min_element(a.begin() + i, a.end()), *max_element(b.begin() + j, b.end()));
    return 0;
}

一定要考虑两种特殊情况，不然满分 25 分只能得 18 分。
